	},

\glsxtrnewsymbol[
	description={%

	},

	description={%

	},

\glsxtrnewsymbol[
	description={%
		电梯外的呼叫请求集合
	},
	unit={\si{(optional) unit}}
]{calle}{\ensuremath{C}}

		$t$时刻请求$f(t)$楼层时该楼层的呼叫请求集合
	},

	},
	unit={\si{(optional) unit}}
]{rft}{\ensuremath{R_{f(t)}}}

\glsxtrnewsymbol[
	description={%
		在$t$时刻，

	},
	unit={\si{(optional) unit}}

% 按键模块

\glsxtrnewsymbol[
	description={%
		表示按键状态的二进制数码，
		$K(t)=(k_{t-3}\,  k_{t-2}\, k_{t-1} \,k_t)_2$
		其中$k_i = 1(i=t,t-1,t-2,t-3)$表示按键释放
	},
	unit={\si{(optional) unit}}
]{kt}{\ensuremath{K(t)}}

\glsxtrnewsymbol[
	description={%
		$t$时刻之前按键状态成功更新的次数
	},
	unit={\si{(optional) unit}}
]{nt}{\ensuremath{N(t)}}

\glsxtrnewsymbol[
	description={%
		$t$时刻的按键状态，
		上一个状态为$S_{N(t)-1}$
	},
	unit={\si{(optional) unit}}
]{snt}{\ensuremath{S_{N(t)} = 0}}

\glsxtrnewsymbol[
	description={%
		$S_t = (S_{N(t)-1}, S_{N(t)})_2$,
		构造数码储存两个按键状态
	},
	unit={\si{(optional) unit}}
]{st}{\ensuremath{S_t}}
